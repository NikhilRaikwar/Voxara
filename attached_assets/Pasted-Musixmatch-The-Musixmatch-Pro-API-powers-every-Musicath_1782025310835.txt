JavaScript
Copy
// Search for a track and fetch its synced lyrics
const BASE = "https://api.musixmatch.com/ws/1.1";
const KEY  = process.env.MXM_KEY;

// 1. Find the track
const search = await fetch(
  `${BASE}/track.search?apikey=${KEY}&q_track=Bohemian+Rhapsody&q_artist=Queen&page_size=1`
);
const { track_list } = (await search.json()).message.body;
const trackId = track_list[0].track.track_id;

// 2. Fetch word-level synced lyrics
const richsync = await fetch(
  `${BASE}/track.richsync.get?apikey=${KEY}&track_id=${trackId}`
);
const { richsync: rs } = (await richsync.json()).message.body;
console.log(JSON.parse(rs.richsync_body));
